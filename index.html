Splitter / Joiner Hello World

String text = "guava,, kiwi, mango,";
Splitter.on(',')
        .split(text);
        // ==> "guava", "", " kiwi", " mango", ""

String text = "guava,, kiwi, mango,";
Splitter.on(',')
        .omitEmptyStrings()
        .split(text);
        // ==> "guava", " kiwi", " mango"

String text = "guava,, kiwi, mango,";
Splitter.on(',')
        .omitEmptyStrings()
        .trimResults()
        .split(text);
        // ==> "guava", "kiwi", "mango"

String text = "guava,, kiwi, mango,";
Splitter.on(',')
        .omitEmptyStrings()
        .trimResults(CharMatcher.WHITESPACE
                .and(CharMatcher.anyOf("aeiou")))
        .split(text);
        // ==> "guav", "kiw", "mang"

String[] fruits = new String[] {"guava", null, "kiwi", "mango"};
Joiner.on(", ")
        .join(fruits);
        // ==> "guava, null, kiwi, mango"

String[] fruits = new String[] {"guava", null, "kiwi", "mango"};
Joiner.on(", ")
        .skipNulls()
        .join(fruits);
        // ==> "guava, kiwi, mango"

String[] fruits = new String[] {"guava", null, "kiwi", "mango"};
Joiner.on(", ")
        .useForNull("<unknown>")
        .join(fruits);
        // ==> "guava, <unknown>, kiwi, mango"

  • Trim arbitrary characters

    String generateNickname(Name name) {
        int nBytes = name.getFirst().length()
            + name.getLast().length();
